3028template<typename Block_, typename Allocator_>
3029[[nodiscard]] constexpr bool operator<(const dynamic_bitset<Block_, Allocator_>& lhs,
3030 const dynamic_bitset<Block_, Allocator_>& rhs)
3031{
3032 using size_type = typename dynamic_bitset<Block_, Allocator_>::size_type;
3033 using block_type = typename dynamic_bitset<Block_, Allocator_>::block_type;
3034 const size_type lhs_size = lhs.size();
3035 const size_type rhs_size = rhs.size();
3036 const size_type lhs_blocks_size = lhs.m_blocks.size();
3037 const size_type rhs_blocks_size = rhs.m_blocks.size();
3038
3039 if(lhs_size == rhs_size)
3040 {
3041 // if comparison of two empty bitsets
3042 if(lhs_size == 0)
3043 {
3044 return false;
3045 }
3046
3047 for(size_type i = lhs_blocks_size - 1; i > 0; --i)
3048 {
3049 if(lhs.m_blocks[i] != rhs.m_blocks[i])
3050 {
3051 return lhs.m_blocks[i] < rhs.m_blocks[i];
3052 }
3053 }
3054 return lhs.m_blocks[0] < rhs.m_blocks[0];
3055 }
3056
3057 // empty bitset inferior to 0-only bitset
3058 if(lhs_size == 0)
3059 {
3060 return true;
3061 }
3062 if(rhs_size == 0)
3063 {
3064 return false;
3065 }
3066
3067 const bool rhs_longer = rhs_size > lhs_size;
3068 const dynamic_bitset<Block_, Allocator_>& longest_bitset = rhs_longer ? rhs : lhs;
3069 const size_type longest_blocks_size = std::max(lhs_blocks_size, rhs_blocks_size);
3070 const size_type shortest_blocks_size = std::min(lhs_blocks_size, rhs_blocks_size);
3071 for(size_type i = longest_blocks_size - 1; i >= shortest_blocks_size; --i)
3072 {
3073 if(longest_bitset.m_blocks[i] != block_type(0))
3074 {
3075 return rhs_longer;
3076 }
3077 }
3078
3079 for(size_type i = shortest_blocks_size - 1; i > 0; --i)
3080 {
3081 if(lhs.m_blocks[i] != rhs.m_blocks[i])
3082 {
3083 return lhs.m_blocks[i] < rhs.m_blocks[i];
3084 }
3085 }
3086 if(lhs.m_blocks[0] != rhs.m_blocks[0])
3087 {
3088 return lhs.m_blocks[0] < rhs.m_blocks[0];
3089 }
3090 return lhs_size < rhs_size;
3091}
